\section{Inside Corner Visualization}

Test path with inside corners: \texttt{101100}

\subsection{Default (no corners shown)}
\begin{center}
\begin{tikzpicture}
\lpDeclarePath{Test1}{101100}
\drawGrid{Test1}
\drawLatticePath[thick, blue]{Test1}
\end{tikzpicture}
\end{center}

\subsection{With inside corners highlighted}
\begin{center}
\begin{tikzpicture}
\lpDeclarePath{Test2}{101100}
\drawGrid{Test2}
\drawLatticePath[thick, blue, lplpath/show inside corners]{Test2}
\end{tikzpicture}
\end{center}

\subsection{With both upmarks and inside corners}
\begin{center}
\begin{tikzpicture}
\lpDeclarePath{Test3}{101100}
\drawGrid{Test3}
\drawLatticePath[thick, blue, lplpath/label upmarks, lplpath/show inside corners]{Test3}
\end{tikzpicture}
\end{center}
